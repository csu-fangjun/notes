39function equalArrays(a, b) {
40  if (a === b) {
41    return true;
42  }
43
44  if (a.length != b.length) {
45    return false;
46  }
47
48  for (let i = 0; i < a.length; i++) {
49    if (a[i] != b[i]) {
50      return false;
51    }
52  }
53
54  return true;
55}
56
